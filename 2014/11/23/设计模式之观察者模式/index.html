设计模式之观察者模式

发表于 2014-11-23   |   分类于 设计模式   |   
设计模式是编程经验、是对OO设计准则的良好编程实践。灵活运用是重点，切忌生搬硬套。
观察者模式定义
原文：<span style="font-family: georgia, palatino;">The Observer Pattern deﬁnes a one-to-many dependency between objects so that when one object changes state, all of its dependents are The Observer Pattern notiﬁed and updated automatically. </span>

翻译：观察者模式定义了一个对象之间一对多的关系，当其中一个对象的状态改变时，所有依赖于这个对象的其它对象能够得到通知并自动更新。
观察者模式UML

 

应用场景

观察者模式中最重要的是将一对多的依赖关系以抽象的方式定义在比较高的级别，让其保持不变，而在这种一对多关系确定情况下的具体变化状况由具体的子类完成。

示例代码
package com.jsonqiao.observerPattern;

/**
 * 观察者模式中的主题对象
 *
 * Created by json_qiao on 2014/11/23.
 */
public interface Subject {

 /**
 * 添加观察者
 */
 public void addObserver(Observer observer);

 /**
 * 删除观察者
 */
 public void deleteObserver(Observer observer);

 /**
 * 通知所有观察者
 */
 public void notifyObservers();
}

package com.jsonqiao.observerPattern;

import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;

/**
 * 具体的主题
 *
 * Created by json_qiao on 2014/11/23.
 */
public class SubjectA implements Subject {

 private int currentNumber = 0;

 /**默认可存储的观察者初始化个数*/
 private static final int DEFAULT_CAPACITY = 15;

 private List<Observer> observers;

 private void initObservers(){
     observers = new ArrayList<Observer>(DEFAULT_CAPACITY);
 }

 public SubjectA(){
     super();
     initObservers();
 }

 @Override
 public void addObserver(Observer observer) {
     observers.add(observer);
     System.out.println("该主题增加一个观察者：" + observer);
 }

 @Override
 public void deleteObserver(Observer observer) {
     observers.remove(observer);
     System.out.println("该主题移除一个观察者：" + observer);
 }

 @Override
 public void notifyObservers() {
     if(!observers.isEmpty()){
         Iterator<Observer> iterator = observers.iterator();
         while (iterator.hasNext()){
             Observer observer = iterator.next();
             System.out.println("开始更新观察者：" + observer);
            observer.update(getCurrentNumber());
        }
     }
 }

 public void setCurrentNumber(int currentNumber) {
     this.currentNumber = currentNumber;
 }

 public int getCurrentNumber() {
     return currentNumber;
 }
}

package com.jsonqiao.observerPattern;

/**
 * 观察者的抽象接口
 *
 * Created by json_qiao on 2014/11/23.
 */
public interface Observer {

 /**
 * 当前观察者主要关心主题的数据为下面的参数
 *
 * @param currentNumber 主题中的当前数字
 */
 public void update(final int currentNumber);
}

package com.jsonqiao.observerPattern;

/**
 * Created by json_qiao on 2014/11/23.
 */
public class ObserverA implements Observer {

 /**默认依赖的主题*/
 private static final Subject DEFAULT_SUBJECT = new SubjectA();

 private Subject subject = DEFAULT_SUBJECT;

 @Override
 public void update(int currentNumber) {
     System.out.println("该主题的当前数据为：" + currentNumber);
 }

 public void setSubject(Subject subject) {
     this.subject = subject;
 }

 @Override
 public String toString() {
     return "观察者A";
 }
}

package com.jsonqiao.observerPattern;

/**

 /**默认依赖的主题*/

 public void setSubject(Subject subject) {

 @Override

}
